17/// Error types for JSON serialization operations.
18#[derive(Debug, thiserror::Error)]
19pub enum SerializationError {
20    /// Error during I/O operations.
21    #[error("I/O error: {0}")]
22    Io(#[from] std::io::Error),
23
24    /// Error during JSON serialization or deserialization.
25    #[error("JSON error: {0}")]
26    Json(#[from] serde_json::Error),
27
28    /// Error when the file format is incompatible.
29    #[error("Incompatible file format: {0}")]
30    IncompatibleFormat(String),
31
32    /// Error when the project version is unsupported.
33    #[error("Unsupported project version: {0}")]
34    UnsupportedVersion(String),
35
36    /// Error from timeline operations
37    #[error("Timeline error: {0}")]
38    Timeline(#[from] crate::project::timeline::TimelineError),
39}
40
41/// Type alias for serialization operation results.
42pub type Result<T> = std::result::Result<T, SerializationError>;

193#[derive(Debug, Serialize, Deserialize, Clone, Copy, PartialEq)]

205/// Serializable representation of multi-track manager.
206#[derive(Debug, Serialize, Deserialize, Default)]
207struct SerializedMultiTrackManager {
208    /// Serialized track relationships.
209    /// Maps source track ID to a map of target track ID to relationship type.
210    relationships: HashMap<String, HashMap<String, SerializedTrackRelationship>>,
211}

213// SerializedTrackRelationshipとTrackRelationshipの相互変換
214impl From<TrackRelationship> for SerializedTrackRelationship {
215    fn from(relationship: TrackRelationship) -> Self {
216        match relationship {
217            TrackRelationship::Independent => Self::Independent,
218            TrackRelationship::Locked => Self::Locked,
219            TrackRelationship::TimingDependent => Self::TimingDependent,
220            TrackRelationship::VisibilityDependent => Self::VisibilityDependent,
221        }
222    }
223}

225impl From<SerializedTrackRelationship> for TrackRelationship {
226    fn from(relationship: SerializedTrackRelationship) -> Self {
227        match relationship {
228            SerializedTrackRelationship::Independent => Self::Independent,
229            SerializedTrackRelationship::Locked => Self::Locked,
230            SerializedTrackRelationship::TimingDependent => Self::TimingDependent,
231            SerializedTrackRelationship::VisibilityDependent => Self::VisibilityDependent,
232        }
233    }
234}

236// MultiTrackManagerとSerializedMultiTrackManagerの相互変換
237impl From<&MultiTrackManager> for SerializedMultiTrackManager {
238    fn from(manager: &MultiTrackManager) -> Self {
239        let mut relationships = HashMap::new();
240
241        // 全ての依存関係を取得してシリアライズ
242        for (source_id, deps) in manager.get_all_relationships() {
243            let source_id_str = source_id.to_string();
244            let mut target_map = HashMap::new();
245
246            for (target_id, relationship) in deps {
247                let target_id_str = target_id.to_string();
248                // ここで*relationshipを使って参照ではなく値を渡す
249                let serialized_relationship = SerializedTrackRelationship::from(*relationship);
250                target_map.insert(target_id_str, serialized_relationship);
251            }
252
253            relationships.insert(source_id_str, target_map);
254        }
255
256        Self { relationships }
257    }
258}

260/// Serializes a project to JSON and writes it to a file.
261///
262/// # Arguments
263///
264/// * `project` - The project to serialize
265/// * `path` - The path where the serialized project will be saved
266///
267/// # Returns
268///
269/// A `Result` containing `()` if the serialization was successful,
270/// or an error if the operation failed.
271///
272/// # Errors
273///
274/// Returns an error if:
275/// * The file could not be created or written to
276/// * The project could not be serialized to JSON
277pub fn serialize_project(project: &Project, path: &Path) -> Result<()> {
278    // Create the file
279    let file = File::create(path)?;
280    let writer = BufWriter::new(file);
281
282    // Create the serializable project
283    let serialized_project = convert_to_serialized_project(project);
284
285    // Create the project file wrapper
286    let project_file = ProjectFile {
287        metadata: ProjectFileMetadata {
288            version: CURRENT_VERSION.to_string(),
289            file_type: "edv_project".to_string(),
290            created_by: format!("edv {}", env!("CARGO_PKG_VERSION")),
291        },
292        project: serialized_project,
293    };
294
295    // Write the project file to JSON
296    to_writer_pretty(writer, &project_file)?;
297
298    Ok(())
299}

301/// Deserializes a project from a JSON file.
302///
303/// # Arguments
304///
305/// * `path` - The path to the serialized project file
306///
307/// # Returns
308///
309/// A `Result` containing the deserialized `Project` if successful,
310/// or an error if the operation failed.
311///
312/// # Errors
313///
314/// Returns an error if:
315/// * The file could not be read
316/// * The JSON could not be parsed
317/// * The file format is incompatible
318/// * The project version is unsupported
319pub fn deserialize_project(path: &Path) -> Result<Project> {
320    // Open the file
321    let file = File::open(path)?;
322    let reader = BufReader::new(file);
323
324    // Parse the JSON
325    let project_file: ProjectFile = from_reader(reader)?;
326
327    // Check the version
328    if !is_version_compatible(&project_file.metadata.version) {
329        return Err(SerializationError::UnsupportedVersion(
330            project_file.metadata.version,
331        ));
332    }
333
334    // Check the file type
335    if project_file.metadata.file_type != "edv_project" {
336        return Err(SerializationError::IncompatibleFormat(
337            project_file.metadata.file_type,
338        ));
339    }
340
341    // Convert to project
342    let project = convert_from_serialized_project(&project_file.project)?;
343
344    Ok(project)
345}

347/// Converts a `Project` to its serializable representation.
348fn convert_to_serialized_project(project: &Project) -> SerializedProject {
349    // Convert project metadata
350    let serialized_metadata = SerializedProjectMetadata {
351        name: project.project_metadata.name.clone(),
352        created_at: project.project_metadata.created_at.to_rfc3339(),
353        modified_at: project.project_metadata.modified_at.to_rfc3339(),
354        description: project.project_metadata.description.clone(),
355        tags: project.project_metadata.tags.clone(),
356    };
357
358    // Convert timeline
359    let serialized_timeline = convert_to_serialized_timeline(&project.timeline);
360
361    // Convert assets
362    let serialized_assets = project
363        .assets
364        .iter()
365        .map(convert_to_serialized_asset_reference)
366        .collect();
367
368    SerializedProject {
369        id: ProjectId::new().to_string(), // 一時的なIDを生成
370        metadata: serialized_metadata,
371        timeline: serialized_timeline,
372        assets: serialized_assets,
373    }
374}

376/// Converts a `Timeline` to its serializable representation.
377fn convert_to_serialized_timeline(timeline: &Timeline) -> SerializedTimeline {
378    let mut serialized_tracks = Vec::new();
379
380    for track in timeline.get_tracks() {
381        let serialized_track = convert_to_serialized_track(track);
382        serialized_tracks.push(serialized_track);
383    }
384
385    // Get the MultiTrackManager from the timeline and serialize it
386    let track_relationships = SerializedMultiTrackManager::from(timeline.multi_track_manager());
387
388    SerializedTimeline {
389        tracks: serialized_tracks,
390        duration: timeline.duration().as_seconds(),
391        track_relationships,
392    }
393}

444/// Converts a serialized project to a `Project`.
445fn convert_from_serialized_project(serialized: &SerializedProject) -> Result<Project> {
446    // Convert project metadata
447    let project_metadata = convert_from_serialized_project_metadata(&serialized.metadata)?;
448
449    // Convert timeline
450    let timeline = convert_from_serialized_timeline(&serialized.timeline)?;
451
452    // Convert assets
453    let assets = serialized
454        .assets
455        .iter()
456        .map(convert_from_serialized_asset_reference)
457        .collect::<std::result::Result<Vec<_>, _>>()?;
458
459    // Create the project
460    let mut project = Project::new(&project_metadata.name);
461    project.project_metadata = project_metadata;
462    project.timeline = timeline;
463    project.assets = assets;
464
465    Ok(project)
466}

468/// Converts serialized project metadata to `ProjectMetadata`.
469fn convert_from_serialized_project_metadata(
470    serialized: &SerializedProjectMetadata,
471) -> Result<ProjectMetadata> {
472    // Parse timestamps
473    let created_at = chrono::DateTime::parse_from_rfc3339(&serialized.created_at)
474        .map_err(|e| SerializationError::IncompatibleFormat(e.to_string()))?
475        .with_timezone(&chrono::Utc);
476
477    let modified_at = chrono::DateTime::parse_from_rfc3339(&serialized.modified_at)
478        .map_err(|e| SerializationError::IncompatibleFormat(e.to_string()))?
479        .with_timezone(&chrono::Utc);
480
481    Ok(ProjectMetadata {
482        name: serialized.name.clone(),
483        created_at,
484        modified_at,
485        description: serialized.description.clone(),
486        tags: serialized.tags.clone(),
487    })
488}

490/// Converts a serialized timeline to a `Timeline`.
491fn convert_from_serialized_timeline(serialized: &SerializedTimeline) -> Result<Timeline> {
492    let mut timeline = Timeline::new();
493
494    // トラックID文字列と実際のTrackIdの対応を格納するマップ
495    let mut track_id_map = HashMap::new();
496
497    // Add each track
498    for serialized_track in &serialized.tracks {
499        let track_id = convert_from_serialized_track(serialized_track, &mut timeline)?;
500
501        // トラックIDの対応を保存
502        track_id_map.insert(serialized_track.id.clone(), track_id);
503
504        // Add clips to the track
505        for serialized_clip in &serialized_track.clips {
506            let clip = convert_from_serialized_clip(serialized_clip)?;
507            timeline.add_clip(track_id, clip)?;
508        }
509    }
510
511    // マルチトラック関係を復元
512    // すべてのトラックが追加された後でトラック関係を追加する
513    restore_track_relationships(
514        &serialized.track_relationships,
515        &track_id_map,
516        &mut timeline,
517    )?;
518
519    Ok(timeline)
520}

522/// マルチトラック関係を復元する補助関数
523///
524/// # Arguments
525///
526/// * `serialized_manager` - シリアライズされたマルチトラックマネージャー
527/// * `track_id_map` - 文字列形式のトラックIDと実際のTrackIdのマッピング
528/// * `timeline` - 関係を追加するタイムライン
529///
530/// # Returns
531///
532/// A `Result` containing `()` if restoration was successful,
533/// or an error if the operation failed.
534///
535/// # Errors
536///
537/// Returns an error if any track in a relationship doesn't exist or if a relationship
538/// would create a circular dependency.
539fn restore_track_relationships(
540    serialized_manager: &SerializedMultiTrackManager,
541    track_id_map: &HashMap<String, TrackId>,
542    timeline: &mut Timeline,
543) -> Result<()> {
544    // 一時的に関係を収集して、あとでまとめて追加する
545    let mut relationships_to_add = Vec::new();
546
547    // シリアライズされた関係を実際のTrackIdにマッピング
548    for (source_id_str, targets) in &serialized_manager.relationships {
549        // 文字列形式のソーストラックIDを実際のTrackIdに変換
550        let Some(source_id) = track_id_map.get(source_id_str) else {
551            return Err(SerializationError::IncompatibleFormat(format!(
552                "Source track ID not found: {source_id_str}"
553            )));
554        };
555
556        // すべてのターゲットトラックの関係も処理
557        for (target_id_str, relationship) in targets {
558            // 文字列形式のターゲットトラックIDを実際のTrackIdに変換
559            let Some(target_id) = track_id_map.get(target_id_str) else {
560                return Err(SerializationError::IncompatibleFormat(format!(
561                    "Target track ID not found: {target_id_str}"
562                )));
563            };
564
565            // SerializedTrackRelationshipからTrackRelationshipへ変換
566            let track_relationship = TrackRelationship::from(*relationship);
567
568            // 関係を収集（あとで追加）
569            relationships_to_add.push((*source_id, *target_id, track_relationship));
570        }
571    }
572
573    // まず、すべてのトラックが存在するかどうかを確認
574    // （エラーチェック）
575    for (source_id, target_id, _) in &relationships_to_add {
576        if !timeline.has_track(*source_id) {
577            return Err(SerializationError::IncompatibleFormat(format!(
578                "Source track not found in timeline: {source_id}"
579            )));
580        }
581        if !timeline.has_track(*target_id) {
582            return Err(SerializationError::IncompatibleFormat(format!(
583                "Target track not found in timeline: {target_id}"
584            )));
585        }
586    }
587
588    // 関係を追加する前に循環依存関係をチェック
589    // 先に循環依存関係の検出を行い、無効な関係を除外する
590    let mut valid_relationships = Vec::new();
591
592    for (source_id, target_id, relationship) in relationships_to_add {
593        // 循環依存関係の簡易チェック
594        // 注: ここでは参照の問題を避けるため、すべてのトラックに関係を追加する前に
595        // 循環依存関係をチェックする簡易的な方法を使用
596        if source_id == target_id
597            || valid_relationships
598                .iter()
599                .any(|(s, t, _)| *s == target_id && *t == source_id)
600        {
601            eprintln!(
602                "Warning: Skipping potential circular dependency between tracks {source_id} and {target_id}"
603            );
604            continue;
605        }
606
607        valid_relationships.push((source_id, target_id, relationship));
608    }
609
610    // 有効な関係をマルチトラックマネージャーに追加
611    // ベストエフォートアプローチ：エラーが発生しても処理を続行
612    for (source_id, target_id, relationship) in valid_relationships {
613        // トラック関係を追加（失敗しても続行）
614        let result = add_track_relationship_safely(timeline, source_id, target_id, relationship);
615
616        if let Err(err) = result {
617            eprintln!("Warning: Failed to add track relationship: {err}");
618        }
619    }
620
621    Ok(())
622}

624/// トラック関係を安全に追加する補助関数
625///
626/// 既存のコードベースを変更せずに、エラーを無視して関係を追加する
627/// タイムラインの参照問題を回避するための特別なアプローチ
628fn add_track_relationship_safely(
629    timeline: &mut Timeline,
630    source_id: TrackId,
631    target_id: TrackId,
632    relationship: TrackRelationship,
633) -> Result<()> {
634    // ダミーのタイムラインを作成する代わりに、トラックの存在を確認した後、
635    // MultiTrackManagerの内部メソッドを使用してリレーションシップを安全に追加
636
637    // 両方のトラックが存在することを確認（冗長だがより明示的）
638    if !timeline.has_track(source_id) || !timeline.has_track(target_id) {
639        return Err(SerializationError::IncompatibleFormat(format!(
640            "Tracks not found: {source_id} or {target_id}"
641        )));
642    }
643
644    // タイムラインのクローンを先に作成して参照問題を回避
645    let timeline_copy = timeline.clone();
646
647    // 直接関係を追加（CircularDependencyエラーは無視）
648    let multi_track = timeline.multi_track_manager_mut();
649
650    // add_relationshipを呼び出し、タイムラインの参照渡しに注意
651    match multi_track.add_relationship(
652        source_id,
653        target_id,
654        relationship,
655        &timeline_copy, // 事前に作成したクローン
656    ) {
657        Ok(()) => Ok(()),
658        Err(e) => {
659            // CircularDependencyエラーの場合は警告を出して続行
660            // このアプローチでは、タイムラインの参照がずれるリスクを抑えつつ、
661            // 可能な限り多くの関係を復元できる
662            eprintln!("Warning when adding relationship: {e}");
663            Err(SerializationError::IncompatibleFormat(format!(
664                "Failed to add track relationship: {e}"
665            )))
666        }
667    }
668}

670/// Converts a serialized track to a `Track` and adds it to the timeline.
671fn convert_from_serialized_track(
672    serialized: &SerializedTrack,
673    timeline: &mut Timeline,
674) -> Result<TrackId> {
675    // Parse track kind
676    let kind = match serialized.kind.as_str() {
677        "Video" => TrackKind::Video,
678        "Audio" => TrackKind::Audio,
679        "Subtitle" => TrackKind::Subtitle,
680        unknown => {
681            return Err(SerializationError::IncompatibleFormat(format!(
682                "Unknown track kind: {unknown}"
683            )));
684        }
685    };
686
687    // Add the track to the timeline
688    let track_id = timeline.add_track(kind);
689
690    // Set track properties
691    let track = timeline
692        .get_track_mut(track_id)
693        .ok_or_else(|| SerializationError::IncompatibleFormat("Failed to add track".to_string()))?;
694
695    track.set_name(&serialized.name);
696    track.set_muted(serialized.muted);
697    track.set_locked(serialized.locked);
698
699    Ok(track_id)
700}

702/// Converts a serialized clip to a `Clip`.
703fn convert_from_serialized_clip(serialized: &SerializedClip) -> Result<Clip> {
704    // Parse clip ID
705    let id = serialized
706        .id
707        .parse()
708        .map_err(|e| SerializationError::IncompatibleFormat(format!("Invalid clip ID: {e}")))?;
709
710    // Parse asset ID
711    let asset_id = serialized
712        .asset_id
713        .parse()
714        .map_err(|e| SerializationError::IncompatibleFormat(format!("Invalid asset ID: {e}")))?;
715
716    // Create the clip
717    Ok(Clip::new(
718        id,
719        asset_id,
720        TimePosition::from_seconds(serialized.position),
721        Duration::from_seconds(serialized.duration),
722        TimePosition::from_seconds(serialized.source_start),
723        TimePosition::from_seconds(serialized.source_end),
724    ))
725}
726
727/// Converts a serialized asset reference to an `AssetReference`.
728fn convert_from_serialized_asset_reference(
729    serialized: &SerializedAssetReference,
730) -> Result<crate::project::AssetReference> {
731    // Parse asset ID
732    let id = serialized
733        .id
734        .parse()
735        .map_err(|e| SerializationError::IncompatibleFormat(format!("Invalid asset ID: {e}")))?;
736
737    // Create asset metadata
738    let metadata = crate::project::AssetMetadata {
739        duration: serialized.metadata.duration.map(Duration::from_seconds),
740        dimensions: serialized.metadata.dimensions.map(|[w, h]| (w, h)),
741        asset_type: serialized.metadata.asset_type.clone(),
742        extra: serialized.metadata.extra.clone(),
743    };
744
745    // Create the asset reference
746    Ok(crate::project::AssetReference {
747        id,
748        path: std::path::PathBuf::from(&serialized.path),
749        metadata,
750    })
751}

753/// Checks if the given version is compatible with the current implementation.
754fn is_version_compatible(version: &str) -> bool {
755    // Parse the version
756    let version_parts: Vec<&str> = version.split('.').collect();
757    if version_parts.len() != 3 {
758        return false;
759    }
760
761    // Parse the major version
762    let major = version_parts[0].parse::<u32>().unwrap_or(0);
763
764    // Currently, we only support version 1.x.x
765    major == 1
766}

768/// Converts a serialized timeline to a `Timeline`.
769fn deserialize_timeline(serialized: &SerializedTimeline) -> Result<Timeline> {
770    convert_from_serialized_timeline(serialized)
771}

773/// Deserializes a track from its serialized representation and adds it to the timeline.
774fn deserialize_track(timeline: &mut Timeline, serialized: SerializedTrack) -> Result<TrackId> {
775    convert_from_serialized_track(&serialized, timeline)
776}
