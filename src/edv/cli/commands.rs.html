22/// Trait that all commands must implement.
23///
24/// This trait defines the interface for all commands in the application.
25/// Each command must provide its name, description, and an execution method
26/// that performs the command's actual functionality.
27pub trait Command: Send + Sync + Debug {
28    /// Gets the name of the command.
29    ///
30    /// This name is used for command registration and lookup.
31    fn name(&self) -> &str;
32
33    /// Gets a human-readable description of the command.
34    ///
35    /// This description is used for help text and documentation.
36    fn description(&self) -> &str;
37
38    /// Gets usage examples for the command.
39    ///
40    /// These examples are displayed in help text to guide users.
41    fn usage(&self) -> &str;
42
43    /// Executes the command with the given context and arguments.
44    ///
45    /// # Arguments
46    ///
47    /// * `context` - The execution context containing configuration and services
48    /// * `args` - Command-specific arguments parsed from the command line
49    ///
50    /// # Returns
51    ///
52    /// `Result<()>` indicating success or failure of the command.
53    fn execute(&self, context: &Context, args: &[String]) -> Result<()>;
54}

56/// Registry for managing available commands.
57///
58/// The command registry maintains a collection of all available commands
59/// and provides methods for registering, looking up, and listing commands.
60#[derive(Debug, Default)]
61pub struct CommandRegistry {
62    /// Map of command names to command implementations
63    commands: HashMap<String, Box<dyn Command>>,
64}
65
66impl CommandRegistry {
67    /// Creates a new, empty command registry.
68    ///
69    /// # Returns
70    ///
71    /// A new `CommandRegistry` instance.
72    #[must_use]
73    pub fn new() -> Self {
74        Self {
75            commands: HashMap::new(),
76        }
77    }
78
79    /// Registers a command with the registry.
80    ///
81    /// # Arguments
82    ///
83    /// * `command` - The command to register
84    ///
85    /// # Returns
86    ///
87    /// `Result<()>` indicating success or whether the command name was already registered.
88    ///
89    /// # Errors
90    ///
91    /// Returns an error if a command with the same name is already registered.
92    pub fn register(&mut self, command: Box<dyn Command>) -> Result<()> {
93        let name = command.name().to_string();
94
95        if self.commands.contains_key(&name) {
96            return Err(Error::DuplicateCommand(name));
97        }
98
99        self.commands.insert(name, command);
100        Ok(())
101    }
102
103    /// Gets a command by name.
104    ///
105    /// # Arguments
106    ///
107    /// * `name` - The name of the command to retrieve
108    ///
109    /// # Returns
110    ///
111    /// `Result<&dyn Command>` containing a reference to the command if found.
112    ///
113    /// # Errors
114    ///
115    /// Returns an error if no command with the given name is registered.
116    pub fn get(&self, name: &str) -> Result<&dyn Command> {
117        self.commands
118            .get(name)
119            .map(AsRef::as_ref)
120            .ok_or_else(|| Error::UnknownCommand(name.to_string()))
121    }
122
123    /// Gets a list of all registered commands.
124    ///
125    /// # Returns
126    ///
127    /// A vector of references to all registered commands.
128    #[must_use]
129    pub fn list(&self) -> Vec<&dyn Command> {
130        self.commands.values().map(AsRef::as_ref).collect()
131    }
132
133    /// Gets a list of all command names.
134    ///
135    /// # Returns
136    ///
137    /// A vector of command names.
138    #[must_use]
139    pub fn command_names(&self) -> Vec<String> {
140        self.commands.keys().cloned().collect()
141    }
142
143    /// Checks if a command with the given name is registered.
144    ///
145    /// # Arguments
146    ///
147    /// * `name` - The name of the command to check
148    ///
149    /// # Returns
150    ///
151    /// `true` if a command with the given name is registered, `false` otherwise.
152    #[must_use]
153    pub fn has_command(&self, name: &str) -> bool {
154        self.commands.contains_key(name)
155    }
156
157    /// Gets the number of registered commands.
158    ///
159    /// # Returns
160    ///
161    /// The number of commands in the registry.
162    #[must_use]
163    pub fn command_count(&self) -> usize {
164        self.commands.len()
165    }
166}

216impl Command for InfoCommand {
217    fn name(&self) -> &str {
218        "info"
219    }
220
221    fn description(&self) -> &str {
222        "Display information about a media file"
223    }
224
225    fn usage(&self) -> &str {
226        "info <file_path> [--detailed]"
227    }
228
229    fn execute(&self, context: &Context, args: &[String]) -> Result<()> {
230        if args.is_empty() {
231            return Err(Error::MissingArgument("file path".to_string()));
232        }
233
234        let file_path = &args[0];
235        let path = Path::new(file_path);
236
237        // Check if file exists
238        if !path.exists() {
239            return Err(Error::InvalidPath(format!("File not found: {}", file_path)));
240        }
241
242        // Get basic file information
243        context
244            .logger
245            .info(&format!("File information for: {}", file_path));
246
247        if let Ok(metadata) = fs::metadata(path) {
248            let size_bytes = metadata.len();
249            let size_formatted = format_file_size(size_bytes);
250
251            context.logger.info(&format!("File exists: Yes"));
252            context
253                .logger
254                .info(&format!("Size: {} ({} bytes)", size_formatted, size_bytes));
255
256            // Get and format modification time
257            if let Ok(modified) = metadata.modified() {
258                let modified: DateTime<Utc> = modified.into();
259                context.logger.info(&format!(
260                    "Modified: {}",
261                    modified.format("%Y-%m-%d %H:%M:%S UTC")
262                ));
263            }
264
265            // Get MIME type (if can be guessed)
266            let mime = MimeGuess::from_path(path).first_or_octet_stream();
267            context.logger.info(&format!("Type: {}", mime));
268
269            // Check if it's a media file
270            let is_media_file = mime.type_().as_str().starts_with("video")
271                || mime.type_().as_str().starts_with("audio")
272                || mime.type_().as_str().starts_with("image");
273
274            // Show if it's a directory
275            if metadata.is_dir() {
276                context.logger.info("Type: Directory");
277            } else if metadata.is_file() {
278                context.logger.info("Type: Regular file");
279            }
280
281            // Show detailed information if requested
282            let detailed = args.len() > 1 && (args[1] == "--detailed" || args[1] == "-d");
283
284            // If it's a media file and FFmpeg is available, get media info
285            if is_media_file {
286                match FFmpeg::detect() {
287                    Ok(ffmpeg) => {
288                        context.logger.info("Media Information:");
289
290                        match ffmpeg.get_media_info(path) {
291                            Ok(media_info) => {
292                                display_media_info(&media_info, context, detailed)?;
293                            }
294                            Err(e) => {
295                                context
296                                    .logger
297                                    .warning(&format!("Could not retrieve media info: {e}"));
298                            }
299                        }
300                    }
301                    Err(e) => {
302                        context
303                            .logger
304                            .warning(&format!("FFmpeg not available: {e}"));
305                        context
306                            .logger
307                            .warning("Media information cannot be displayed without FFmpeg.");
308                    }
309                }
310            } else if detailed {
311                context.logger
312                    .info("Note: This is not a media file, so no media-specific information is available.");
313            }
314        } else {
315            return Err(Error::InvalidPath(format!(
316                "Could not read file metadata: {}",
317                file_path
318            )));
319        }
320
321        context.logger.info("Info command executed successfully");
322        Ok(())
323    }
324}

326/// Displays media information.
327///
328/// # Arguments
329///
330/// * `media_info` - The media information to display
331/// * `context` - The execution context
332/// * `detailed` - Whether to display detailed information
333///
334/// # Returns
335///
336/// A Result indicating success or failure
337///
338/// # Errors
339///
340/// Returns an error if the operation fails
341fn display_media_info(media_info: &MediaInfo, context: &Context, detailed: bool) -> Result<()> {
342    // Display format info
343    let format = &media_info.format;
344    context
345        .logger
346        .info(&format!("  Format: {}", format.format_long_name));
347
348    // Display duration
349    if let Some(duration) = media_info.duration_seconds() {
350        let duration_fmt = format_duration(duration);
351        context
352            .logger
353            .info(&format!("  Duration: {}", duration_fmt));
354    }
355
356    // Display bit rate
357    if let Some(bit_rate) = media_info.bit_rate() {
358        context
359            .logger
360            .info(&format!("  Bit Rate: {} kb/s", bit_rate / 1000));
361    }
362
363    // Display video streams
364    let video_streams = media_info.video_streams();
365    if !video_streams.is_empty() {
366        context
367            .logger
368            .info(&format!("  Video Streams: {}", video_streams.len()));
369
370        for (i, stream) in video_streams.iter().enumerate() {
371            context
372                .logger
373                .info(&format!("    Stream #{}: {}", i, stream.codec_long_name));
374
375            if let (Some(width), Some(height)) = (stream.width, stream.height) {
376                context
377                    .logger
378                    .info(&format!("      Resolution: {}x{}", width, height));
379            }
380
381            if let Some(frame_rate) = &stream.frame_rate {
382                if let Ok((num, den)) = parse_frame_rate(frame_rate) {
383                    let fps = num as f64 / den as f64;
384                    context
385                        .logger
386                        .info(&format!("      Frame Rate: {:.2} fps", fps));
387                }
388            }
389
390            if detailed {
391                if let Some(pixel_format) = &stream.pixel_format {
392                    context
393                        .logger
394                        .info(&format!("      Pixel Format: {}", pixel_format));
395                }
396
397                if let Some(bit_rate) = &stream.bit_rate {
398                    if let Ok(br) = bit_rate.parse::<u64>() {
399                        context
400                            .logger
401                            .info(&format!("      Bit Rate: {} kb/s", br / 1000));
402                    }
403                }
404
405                // Display tags if available
406                if let Some(tags) = &stream.tags {
407                    context.logger.info("      Tags:");
408                    for (key, value) in tags {
409                        context.logger.info(&format!("        {}: {}", key, value));
410                    }
411                }
412            }
413        }
414    }
415
416    // Display audio streams
417    let audio_streams = media_info.audio_streams();
418    if !audio_streams.is_empty() {
419        context
420            .logger
421            .info(&format!("  Audio Streams: {}", audio_streams.len()));
422
423        for (i, stream) in audio_streams.iter().enumerate() {
424            context
425                .logger
426                .info(&format!("    Stream #{}: {}", i, stream.codec_long_name));
427
428            if let Some(sample_rate) = &stream.sample_rate {
429                context
430                    .logger
431                    .info(&format!("      Sample Rate: {} Hz", sample_rate));
432            }
433
434            if let Some(channels) = stream.channels {
435                context
436                    .logger
437                    .info(&format!("      Channels: {}", channels));
438
439                if let Some(channel_layout) = &stream.channel_layout {
440                    context
441                        .logger
442                        .info(&format!("      Channel Layout: {}", channel_layout));
443                }
444            }
445
446            if detailed {
447                if let Some(bit_rate) = &stream.bit_rate {
448                    if let Ok(br) = bit_rate.parse::<u64>() {
449                        context
450                            .logger
451                            .info(&format!("      Bit Rate: {} kb/s", br / 1000));
452                    }
453                }
454
455                // Display tags if available
456                if let Some(tags) = &stream.tags {
457                    context.logger.info("      Tags:");
458                    for (key, value) in tags {
459                        context.logger.info(&format!("        {}: {}", key, value));
460                    }
461                }
462            }
463        }
464    }
465
466    // Display subtitle streams if detailed mode
467    if detailed {
468        let subtitle_streams = media_info.subtitle_streams();
469        if !subtitle_streams.is_empty() {
470            context
471                .logger
472                .info(&format!("  Subtitle Streams: {}", subtitle_streams.len()));
473
474            for (i, stream) in subtitle_streams.iter().enumerate() {
475                context
476                    .logger
477                    .info(&format!("    Stream #{}: {}", i, stream.codec_long_name));
478
479                if let Some(tags) = &stream.tags {
480                    context.logger.info("      Tags:");
481                    for (key, value) in tags {
482                        context.logger.info(&format!("        {}: {}", key, value));
483                    }
484                }
485            }
486        }
487
488        // Display format tags if available
489        if let Some(tags) = &format.tags {
490            context.logger.info("  Format Tags:");
491            for (key, value) in tags {
492                context.logger.info(&format!("    {}: {}", key, value));
493            }
494        }
495    }
496
497    Ok(())
498}

500/// Formats a file size into a human-readable string.
501///
502/// # Arguments
503///
504/// * `size` - The file size in bytes
505///
506/// # Returns
507///
508/// A human-readable string representing the file size
509fn format_file_size(size: u64) -> String {
510    const KB: u64 = 1024;
511    const MB: u64 = KB * 1024;
512    const GB: u64 = MB * 1024;
513
514    if size >= GB {
515        format!("{:.2} GB", size as f64 / GB as f64)
516    } else if size >= MB {
517        format!("{:.2} MB", size as f64 / MB as f64)
518    } else if size >= KB {
519        format!("{:.2} KB", size as f64 / KB as f64)
520    } else {
521        format!("{} bytes", size)
522    }
523}

525/// Formats a duration in seconds to a human-readable string.
526///
527/// # Arguments
528///
529/// * `seconds` - Duration in seconds
530///
531/// # Returns
532///
533/// A formatted string representing the duration (e.g., "1h 23m 45s")
534fn format_duration(seconds: f64) -> String {
535    let hours = (seconds / 3600.0).floor() as u64;
536    let minutes = ((seconds % 3600.0) / 60.0).floor() as u64;
537    let secs = (seconds % 60.0).floor() as u64;
538
539    if hours > 0 {
540        format!("{}h {}m {}s", hours, minutes, secs)
541    } else if minutes > 0 {
542        format!("{}m {}s", minutes, secs)
543    } else {
544        format!("{:.2}s", seconds)
545    }
546}

548/// Parses a frame rate string in the format "num/den".
549///
550/// # Arguments
551///
552/// * `frame_rate` - The frame rate string
553///
554/// # Returns
555///
556/// A Result containing a tuple of (numerator, denominator)
557///
558/// # Errors
559///
560/// Returns an error if the string cannot be parsed
561fn parse_frame_rate(frame_rate: &str) -> Result<(u64, u64)> {
562    let parts: Vec<&str> = frame_rate.split('/').collect();
563    if parts.len() != 2 {
564        return Err(Error::InvalidArgument(format!(
565            "Invalid frame rate format: {frame_rate}"
566        )));
567    }
568
569    let num = parts[0]
570        .parse::<u64>()
571        .map_err(|e| Error::InvalidArgument(format!("Invalid frame rate numerator: {e}")))?;
572
573    let den = parts[1]
574        .parse::<u64>()
575        .map_err(|e| Error::InvalidArgument(format!("Invalid frame rate denominator: {e}")))?;
576
577    if den == 0 {
578        return Err(Error::InvalidArgument(
579            "Frame rate denominator cannot be zero".to_string(),
580        ));
581    }
582
583    Ok((num, den))
584}

586/// Command for trimming a video file.
587///
588/// This command trims a video file to a specified start and end time,
589/// creating a new file with the trimmed content. It supports both
590/// stream copy (fast) and re-encoding (higher quality for precise cuts).
591#[derive(Debug)]
592pub struct TrimCommand;
593
594impl TrimCommand {
595    /// Creates a new trim command.
596    #[must_use]
597    pub fn new() -> Self {
598        Self
599    }
600
601    /// Checks if the given file exists and is readable.
602    ///
603    /// # Arguments
604    ///
605    /// * `path` - Path to the file to check
606    ///
607    /// # Returns
608    ///
609    /// `Result<()>` indicating success or failure.
610    fn check_input_file(&self, path: &str) -> Result<()> {
611        let path = Path::new(path);
612        if !path.exists() {
613            return Err(Error::InvalidArgument(format!(
614                "Input file does not exist: {path:?}"
615            )));
616        }
617        if !path.is_file() {
618            return Err(Error::InvalidArgument(format!(
619                "Not a regular file: {path:?}"
620            )));
621        }
622        Ok(())
623    }
624
625    /// Extracts and validates time position arguments.
626    ///
627    /// # Arguments
628    ///
629    /// * `start_str` - Optional start time string
630    /// * `end_str` - Optional end time string
631    /// * `duration` - Media duration for validation
632    ///
633    /// # Returns
634    ///
635    /// Result with tuple of (start_pos, end_pos) as TimePosition objects
636    fn extract_time_positions(
637        &self,
638        start_str: Option<&str>,
639        end_str: Option<&str>,
640        duration: f64,
641    ) -> Result<(
642        crate::utility::time::TimePosition,
643        crate::utility::time::TimePosition,
644    )> {
645        use crate::utility::time::TimePosition;
646
647        // Parse start time (default to 0)
648        let start_pos = if let Some(start) = start_str {
649            TimePosition::parse(start)
650                .map_err(|e| Error::InvalidArgument(format!("Invalid start time: {e}")))?
651        } else {
652            TimePosition::zero()
653        };
654
655        // Parse end time (default to duration)
656        let end_pos = if let Some(end) = end_str {
657            TimePosition::parse(end)
658                .map_err(|e| Error::InvalidArgument(format!("Invalid end time: {e}")))?
659        } else {
660            TimePosition::from_seconds(duration)
661        };
662
663        // Validate time positions
664        if start_pos.as_seconds() >= end_pos.as_seconds() {
665            return Err(Error::InvalidArgument(
666                "Start time must be less than end time".to_string(),
667            ));
668        }
669
670        if end_pos.as_seconds() > duration {
671            return Err(Error::InvalidArgument(format!(
672                "End time ({}) exceeds media duration ({})",
673                end_pos,
674                format_duration(duration)
675            )));
676        }
677
678        Ok((start_pos, end_pos))
679    }
680}

682impl Command for TrimCommand {
683    fn name(&self) -> &str {
684        "trim"
685    }
686
687    fn description(&self) -> &str {
688        "Trims a video file to specified start and end times"
689    }
690
691    fn usage(&self) -> &str {
692        "trim --input <input_file> --output <output_file> [--start <time>] [--end <time>] [--recompress]"
693    }
694
695    fn execute(&self, context: &Context, args: &[String]) -> Result<()> {
696        if args.len() < 2 {
697            return Err(Error::InvalidArgument(
698                "Missing required arguments. Use --help for usage information.".to_string(),
699            ));
700        }
701
702        let input_file = &args[0];
703        let output_file = &args[1];
704
705        // Parse remaining arguments
706        let mut start_time = None;
707        let mut end_time = None;
708        let mut recompress = false;
709        let mut show_frames = true; // 
710
711        let mut i = 2;
712        while i < args.len() {
713            match args[i].as_str() {
714                "--start" => {
715                    if i + 1 < args.len() {
716                        start_time = Some(args[i + 1].as_str());
717                        i += 2;
718                    } else {
719                        return Err(Error::InvalidArgument(
720                            "--start requires a value".to_string(),
721                        ));
722                    }
723                }
724                "--end" => {
725                    if i + 1 < args.len() {
726                        end_time = Some(args[i + 1].as_str());
727                        i += 2;
728                    } else {
729                        return Err(Error::InvalidArgument("--end requires a value".to_string()));
730                    }
731                }
732                "--recompress" => {
733                    recompress = true;
734                    i += 1;
735                }
736                "--no-frames" => {
737                    show_frames = false;
738                    i += 1;
739                }
740                _ => {
741                    return Err(Error::InvalidArgument(format!(
742                        "Unknown argument: {}",
743                        args[i]
744                    )));
745                }
746            }
747        }
748
749        // Validate input file
750        self.check_input_file(input_file)?;
751
752        // Initialize FFmpeg
753        let ffmpeg = FFmpeg::detect().map_err(|e| Error::FFmpegError(e.to_string()))?;
754
755        // Get media information to validate times
756        context
757            .logger
758            .info(&format!("Analyzing media file: {input_file}"));
759        let media_info = ffmpeg
760            .get_media_info(input_file)
761            .map_err(|e| Error::FFmpegError(format!("Failed to get media info: {e}")))?;
762
763        // Get duration from format section
764        let duration_str = media_info.format.duration.as_ref().ok_or_else(|| {
765            Error::CommandExecution("Media file has no duration information".to_string())
766        })?;
767
768        let duration = duration_str
769            .parse::<f64>()
770            .map_err(|_| Error::CommandExecution("Invalid duration in media info".to_string()))?;
771
772        // Extract and validate time positions
773        let (start_pos, end_pos) = self.extract_time_positions(start_time, end_time, duration)?;
774
775        // Build FFmpeg command for trimming
776        context.logger.info(&format!(
777            "Trimming from {} to {} (duration: {})",
778            start_pos,
779            end_pos,
780            format_duration(end_pos.as_seconds() - start_pos.as_seconds())
781        ));
782
783        let mut ffmpeg_cmd = ffmpeg.command();
784
785        // Add input options
786        ffmpeg_cmd.add_input_option("-ss", &start_pos.to_string());
787
788        // Add input file
789        ffmpeg_cmd.add_input(input_file);
790
791        // Add duration (more precise than end time)
792        let trim_duration = end_pos.as_seconds() - start_pos.as_seconds();
793        ffmpeg_cmd.add_output_option("-t", &trim_duration.to_string());
794
795        // Configure output options
796        if !recompress {
797            // Use stream copy for faster processing
798            ffmpeg_cmd.add_output_option("-c", "copy");
799        }
800
801        // Set output file
802        ffmpeg_cmd.set_output(output_file);
803
804        // Create progress reporter with total duration in seconds
805        let mut progress = ProgressReporter::new(trim_duration as usize, "Trimming video", true);
806
807        // Execute FFmpeg command
808        ffmpeg_cmd
809            .execute_with_progress(|progress_line| {
810                // Process progress updates from FFmpeg
811                if let Some(time_pos) = extract_time_from_progress(progress_line) {
812                    progress.update(time_pos as usize);
813                }
814            })
815            .map_err(|e| Error::FFmpegError(format!("FFmpeg execution failed: {e}")))?;
816
817        progress.complete();
818        context
819            .logger
820            .info(&format!("Trimmed video saved to: {output_file}"));
821
822        //  - Windows
823        if show_frames && !cfg!(windows) {
824            // Unix
825            ffmpeg_cmd.add_output_option("-vf", "drawtext=text=%{n}/%{nb_frames}:fontcolor=white");
826        }
827
828        // - 
829        ffmpeg_cmd.add_output_option("-stats", "");
830
831        Ok(())
832    }
833}

835/// Extracts time position from FFmpeg progress output.
836///
837/// # Arguments
838///
839/// * `line` - Progress line from FFmpeg
840///
841/// # Returns
842///
843/// Option containing time position in seconds if found
844fn extract_time_from_progress(line: &str) -> Option<f64> {
845    // Extract time from FFmpeg progress line (e.g., "time=00:00:10.00")
846    if let Some(pos) = line.find("time=") {
847        let time_str = &line[pos + 5..];
848        if let Some(end) = time_str.find(' ') {
849            let time_val = &time_str[..end];
850            // Handle different time formats (HH:MM:SS.MS or seconds)
851            if time_val.contains(':') {
852                // Parse time in format HH:MM:SS.MS
853                let parts: Vec<&str> = time_val.split(':').collect();
854                if parts.len() == 3 {
855                    let hours = parts[0].parse::<f64>().unwrap_or(0.0);
856                    let minutes = parts[1].parse::<f64>().unwrap_or(0.0);
857                    let seconds = parts[2].parse::<f64>().unwrap_or(0.0);
858                    return Some(hours * 3600.0 + minutes * 60.0 + seconds);
859                }
860            } else if let Ok(secs) = time_val.parse::<f64>() {
861                return Some(secs);
862            }
863        }
864    }
865    None
866}

996/// Play a video file.
997#[derive(Debug)]
998pub struct PlayCommand;
999
1000impl PlayCommand {
1001    /// Creates a new play command.
1002    #[must_use]
1003    pub fn new() -> Self {
1004        Self
1005    }
1006
1007    /// Validates that the input file exists and is readable.
1008    ///
1009    /// # Arguments
1010    ///
1011    /// * `path` - The path to validate
1012    ///
1013    /// # Returns
1014    ///
1015    /// `Ok(())` if the file is valid, or an `Error` otherwise.
1016    ///
1017    /// # Errors
1018    ///
1019    /// Returns an error if the file does not exist or is not readable.
1020    fn check_input_file(&self, path: &str) -> Result<()> {
1021        let path = Path::new(path);
1022        if !path.exists() {
1023            return Err(Error::InvalidPath(format!(
1024                "Input file does not exist: {}",
1025                path.display()
1026            )));
1027        }
1028
1029        if !path.is_file() {
1030            return Err(Error::InvalidPath(format!(
1031                "Input path is not a file: {}",
1032                path.display()
1033            )));
1034        }
1035
1036        // Try to open the file to check if it's readable
1037        match fs::File::open(path) {
1038            Ok(_) => Ok(()),
1039            Err(e) => Err(Error::InvalidPath(format!(
1040                "Input file is not readable: {}: {}",
1041                path.display(),
1042                e
1043            ))),
1044        }
1045    }
1046
1047    /// Extract and validate time positions for playing.
1048    ///
1049    /// # Arguments
1050    ///
1051    /// * `start_time` - Optional starting time string
1052    /// * `end_time` - Optional ending time string
1053    /// * `duration` - Total duration of the file in seconds
1054    ///
1055    /// # Returns
1056    ///
1057    /// A tuple of (start, end) positions.
1058    ///
1059    /// # Errors
1060    ///
1061    /// Returns an error if the time positions are invalid.
1062    fn extract_time_positions(
1063        &self,
1064        start_time: Option<&str>,
1065        end_time: Option<&str>,
1066        duration: f64,
1067    ) -> Result<(TimePosition, TimePosition)> {
1068        // Default start position is beginning of file
1069        let start_pos = if let Some(start) = start_time {
1070            start
1071                .parse::<TimePosition>()
1072                .map_err(|e| Error::InvalidTimeFormat(format!("Invalid start time: {start}")))?
1073        } else {
1074            TimePosition::from_seconds(0.0)
1075        };
1076
1077        // Default end position is end of file
1078        let end_pos = if let Some(end) = end_time {
1079            end.parse::<TimePosition>()
1080                .map_err(|e| Error::InvalidTimeFormat(format!("Invalid end time: {end}")))?
1081        } else {
1082            TimePosition::from_seconds(duration)
1083        };
1084
1085        // Validate positions
1086        if start_pos.as_seconds() < 0.0 {
1087            return Err(Error::InvalidTimeFormat(
1088                "Start time cannot be negative".to_string(),
1089            ));
1090        }
1091
1092        if end_pos.as_seconds() > duration {
1093            return Err(Error::InvalidTimeFormat(format!(
1094                "End time ({}) exceeds file duration ({})",
1095                end_pos.as_seconds(),
1096                duration
1097            )));
1098        }
1099
1100        if start_pos.as_seconds() >= end_pos.as_seconds() {
1101            return Err(Error::InvalidTimeFormat(
1102                "Start time must be less than end time".to_string(),
1103            ));
1104        }
1105
1106        Ok((start_pos, end_pos))
1107    }
1108}
